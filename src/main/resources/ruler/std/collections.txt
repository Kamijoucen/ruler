fun IsArray(arr) {
    return typeof(arr) == "array";
}

fun IsNotArray(arr) {
    return !IsArray(arr);
}

fun Size(arr) {
    if !IsArray(arr) return null;
    return length(arr);
}

fun IsEmpty(list) {
    if list == null return true;
    if !IsArray(arr) return false;
    return length(arr) == 0;
}

fun IsNotEmpty(list) {
    return !IsEmpty(list);
}

fun Swap(arr, i, j) {
    var temp = arr[i];
    arr[i] = arr[j];
    arr[j] = temp;
}

fun Contains(obj, list) {
    if !IsArray(list) return false;
    for v in list
        if v == obj return true;
    return false;
}

fun In(obj, list) {
    return Contains(obj, list);
}
